	AREA FIBO,CODE,READONLY
         	ENTRY
         	EXPORT __main
__main
        	MOV R2,#10
        	LDR R1,=0X10000000

REP    		LDR R3,[R1]	; R3 = a
       		ADD R1,#04
        	LDR R4,[R1]	; R4 = b
        	ADD R5,R4,R3
        	STR R5,[R1,#04]	; R5 = a+b
        	SUBS R2,#01
        	BNE REP
		NOP
        	END
