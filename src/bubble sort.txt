	AREA BubbleSort,CODE,READONLY
		ENTRY
		EXPORT __main
__main
		MOV R2,#5
Out_loop	
		LDR R1,=0X10000000
		MOV R3,R2
In_loop		
		LDR R4,[R1]
		ADD R1,#4
		LDR R5,[R1]

		CMP R4,R5
		BLO skip	;BHI for desc
		STR R4,[R1]
		STR R5,[R1,#-4]

skip		
		SUBS R3,#01
		BNE In_loop
		SUBS R2,#01
		BNE Out_loop

Stay		B Stay
		END
