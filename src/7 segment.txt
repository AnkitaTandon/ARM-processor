#include<LPC17XX.H>
int delay,count;

int Disp[16]={0X001003F0,0X00100060,0X001005B0,0X001004F0,0X00100660,0X001006D0,
		0X001007D0,0X00100070,0X001007F0,0X001006F0,0X00100770,0X001007C0,0X00100390,0X001005E0,0X00100790,0X00100710};

int main(){
	LPC_PINCON->PINSEL0=0X00000000;		//configure P0.4 to P0.11 as GPIO
	LPC_GPIO0->FIODIR=0X00180FF0;		//configure P0.4 to P0.11 as OUTPUT

	while(1){
		for(count=0;count<16;count++){
		LPC_GPIO0->FIOPIN=Disp[count];

		for(delay=0;delay<90000;delay++);
		}
	}
}
